32 SDL_Surface *load_image( string filename )
33 {
34  //Immagine iniziale
35  SDL_Surface* loadedImage = NULL;
36 
37  //Immagine finale
38  SDL_Surface* optimizedImage = NULL;
39 
40  //Caricamento dell'immagine
41  loadedImage = IMG_Load( filename.c_str() );
42  if( loadedImage != NULL )
43  {
44  //Crea l'immagine finale
45  optimizedImage = SDL_DisplayFormat( loadedImage );
46  if( optimizedImage != NULL )
47  {
48  //Applica la color key
49  SDL_SetColorKey( optimizedImage, SDL_SRCCOLORKEY, SDL_MapRGB( optimizedImage->format, 0, 0xFF, 0xFF ) );
50  }
51 
52  //Rimuovi l'immagine iniziale.
53  SDL_FreeSurface( loadedImage );
54  }
55 
56  //Ritorna l'immagine finale
57  return optimizedImage;
58 }
