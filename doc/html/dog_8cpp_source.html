60 void step_frame()
61 {
62  if( d.frame == 2 )
63  {
64  d.walk_n++;
65  d.frame = 0;
66  }
67  else
68  {
69  if( d.frame == 4 )
70  {
71  d.sniff_n++;
72  d.frame = 3;
73  }
74  else
75  {
76  d.frame++;
77  }
78  }
79 
80  if( d.walk_n > 4 )

86 
87  if( d.sniff_n > 2 )

203 bool dog_intro_animation()
204 {
205  ++d.ani_count;
206  switch( d.ani_count )
207  {
208  case 0 ... 61:
209  step_frame();
210  dog_move();
211  SDL_Delay( 120 );
212  break;
213 
214  case 62 ... 64:
215  d.frame = 5;
216  SDL_Delay( 120 );
217  break;
218 
219  case 65:
220  Mix_PlayChannel( -1, bark, 0 );
221 
222  case 66 ... 70:
223  d.frame = 6;
224  d.x_offset += 5;
225  d.y_offset += 9;
226  break;
227 
228  case 71 ... 75:
229  d.frame = 6;
230  d.x_offset += 5;
231  d.y_offset += 4;
232  break;
233 
234  case 76 ... 90:
235  d.frame = 7;
236  d.x_offset += 5;
237  d.y_offset -= 6;
238  break;
239 
240  default:
241  return false;
242  break;

245  return true;
