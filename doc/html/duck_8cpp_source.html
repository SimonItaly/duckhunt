145 void duck_set_dest( duck &d )
146 {
147  //Sinistra
148  if ( d.x_offset < 5 )
149  {
150  d.angle = 270 + (rand() % 180);
151  }
152  //Destra
153  else if ( d.x_offset + duck_frames[ d.frame ].w > SCREEN_WIDTH )
154  {
155  d.angle = 90 + (rand() % 180);
156  }
157  //Alto
158  else if ( d.y_offset < 5 )
159  {
160  d.angle = 0 + (rand() % 180);
161  }
162  //Basso
163  else if ( d.y_offset + duck_frames[ d.frame ].h > 270 )
164  {
165  d.angle = 180 + (rand() % 180);
166  }
167 
168  //No angoli > 360
169  d.angle = d.angle % 360;
170 }

172 void duck_init( duck &d, int speed )
173 {
174  //Random seed
175  srand( SDL_GetTicks() );
176 
177  //Frame dell'animazione
178  d.frame = 0;
179 
180  //Contatore animazioni
181  d.counter = 0;
182 
183  //Posizione iniziale
184  d.x_offset = rand() % SCREEN_WIDTH;
185  d.y_offset = 200;
186 
187  //Direzione iniziale verso destra
188  d.angle = ( 270 + (rand() % 90) ) % 360;
189 
190  //Velocità, dipendente dal round corrente
191  d.speed = DUCK_SPEED + speed;
192 
193  //Fly away
194  d.away = false;
195 
196  //Inizializza il tipo di papera
197  switch( rand() % 3 )
198  {
199  case 0:
200  d.type = BLACK_DUCK;
201  d.points = 500;
202  break;
203  case 1:
204  d.type = RED_DUCK;
205  d.points = 1500;
206  break;
207  case 2:
208  d.type = BLUE_DUCK;
209  d.points = 1000;
210  break;
211  default:
212  break;
213  }
214 
215  duck_set_clips();
216 }

218 bool duck_shot_anim( duck &d )
219 {
220  //Mostra il frame della papera colpita
221  if(d.frame < 9 || d.counter < 20)
222  {
223  d.frame = 9;
224  d.counter++;
225  }
226  else
227  {
228  //Alterna l'animazione di caduta
229  if(d.y_offset % 10 == 0)
230  {
231  if(d.frame == 11)
232  d.frame = 10;
233  else
234  d.frame = 11;
235  }
236  d.y_offset ++;
237  }
238 
239  apply_surface( d.x_offset, d.y_offset, sprite_sheet[ d.type ], screen, &duck_frames[ d.frame ] );
240  apply_surface( 0, 0, grass_over, screen );
241 
242  //Ritorna true se si trova sotto l'erba
243  if(d.y_offset < 260)
244  return true;
245 
246  return false;
247 }

249 void duck_move( duck &d, bool fly_away )
250 {
251  //Movimento in base all'angolo
252  //La costante double 0.0174532925 permette la conversione
253  //da radianti a gradi di cos() e sin()
254  d.x_offset = d.x_offset + int(d.speed * cos((double)d.angle * 0.0174532925));
255  d.y_offset = d.y_offset + int(d.speed * sin((double)d.angle * 0.0174532925));
256 
257  //Se è volata via procedi fino a -300 per farla sparire dallo schermo
258  if(fly_away)
259  {
260  if(d.y_offset < -300)
261  d.away = true;
262  }
263  else if( d.x_offset < 5 ||
264  d.x_offset + duck_frames[ d.frame ].w > SCREEN_WIDTH ||
265  d.y_offset < 5 ||
266  d.y_offset + duck_frames[ d.frame ].h > 270 )
267  {
268  //Se ha raggiunto la destinazione ricalcola l'angolo
269  duck_set_dest( d );
270  }
271 }

273 void duck_show( duck &d )
274 {
275  //Animazione movimento
276  d.counter++;
277  if(d.counter)
278  {
279  d.counter = 0;
280 
281  d.frame++;
282  switch(d.angle)
283  {
284  case 45 ... 224: //Diagonale 3,4,5
285  {
286  if(d.frame < 3 || d.frame > 5)
287  d.frame = 3;
288  break;
289  }
290  case 225 ... 315: //Verticale 6,7,8
291  {
292  if(d.frame < 6 || d.frame > 8)
293  d.frame = 6;
294  break;
295  }
296  default: //Orizzontale 0,1,2
297  {
298  if(d.frame > 2)
299  d.frame = 0;
300  break;
301  }
302  }
303  }
304 
305  apply_surface( d.x_offset, d.y_offset, sprite_sheet[ d.type ], screen, &duck_frames[ d.frame ] );
306  apply_surface( 0, 0, grass_over, screen );
307 
308  SDL_Delay( 50 );
309 }
