498 int show_duck_miss( bool game_over )
499 {
500  timer game_over_timer;
501  timer_start( game_over_timer );
502 
503  int timer_seconds,
504  max_seconds = 4;
505 
506  if(game_over)
507  max_seconds += 3;
508 
509  dog_lol_init();
510 
511  //Animazione cane con game over
512  do
513  {
514  //Polling degli eventi
515  event_ret = handle_events();
516  if(event_ret != RETURN_NO_EVENT)
517  return event_ret;
518 
519  screen_update();
520 
521  timer_seconds = timer_passed_seconds( game_over_timer );
522 
523  if(game_over)
524  {
525  apply_game_over();
526  if( timer_seconds > 2 )
527  {
528  dog_show_lol( false );
529  }
530  }
531  else
532  {
533  dog_show_lol( timer_seconds > 2 );
534  }
535 
536  apply_surface( 0, 0, grass_over, screen );
537 
538  SDL_Flip( screen );
539  }
540  while(timer_seconds < max_seconds);
541 
542  return RETURN_NO_EVENT;
543 }

560 void screen_update()
561 {
562  SDL_Rect round_rect = { 96, 352, 0, 16 };
563  SDL_Rect duck_rect = { 0, 388, 14, 14 };
564  SDL_Rect pass_rect = { 0, 0, 16, 14 };
565 
566  if( !fly_away )
567  //Sfondo blu
568  SDL_FillRect( screen, &screen->clip_rect, SDL_MapRGB( screen->format, 0x3F, 0xBF, 0xFF ) );
569  else
570  //Sfondo rosa
571  SDL_FillRect( screen, &screen->clip_rect, SDL_MapRGB( screen->format, 0xFF, 0xBF, 0xB3 ) );
572 
573  //Icona papera bianca o rossa
574  for(int i = 0; i < 10; i++)
575  {
576  duck_rect.x = 192 + 16 * i;
577 
578  //Se la papera stata colpita l'icona rossa
579  if(hit_duck[i])
580  SDL_FillRect( screen, &duck_rect, SDL_MapRGB( screen->format, 228, 68, 55 ) );
581  else
582  {
583  //Flash bianco-nero della papera gestito dal timer
584  if(i == cur_duck)
585  {
586  if( ( timer_get_ticks( duck_blink ) / 150 ) % 2 )
587  SDL_FillRect( screen, &duck_rect, SDL_MapRGB( screen->format, 0xFF, 0xFF, 0xFF ) );
588  else
589  SDL_FillRect( screen, &duck_rect, SDL_MapRGB( screen->format, 0, 0, 0 ) );
590  }
591  else
592  SDL_FillRect( screen, &duck_rect, SDL_MapRGB( screen->format, 0xFF, 0xFF, 0xFF ) );
593  }
594  }
595 
596  //Papere per passare il round
597  for(int i = 0; i < pass_ducks; i++)
598  {
599  apply_surface( 192 + (16 * i), 402, pass_lines, screen, &pass_rect);
600  }
601  pass_rect.x = 16;
602  for(int i = pass_ducks; i < 10; i++)
603  {
604  apply_surface( 192 + (16 * i), 402, pass_lines, screen, &pass_rect );
605  }
606 
607  //Interfaccia
608  apply_surface( 0, 0, duck_bg, screen );
609 
610  //Proiettili rimanenti
611  for(int i = 0; i < bullets_amount; i++)

613  apply_surface( 52 + 16 * i, 384, bullet, screen );
614  }
615  //Proiettili sparati (sprite nero)
616  for(int i = bullets_amount; i < 3; i++)

618  SDL_FillRect( screen, 52 + 16 * i, 384, 14, 8, SDL_MapRGB( screen->format, 0, 0, 0 ) );
619  }
620 
621  //Punteggio
622  score_text = TTF_RenderText_Solid( font, ( convert_score( score ) ).c_str(), white_font );
623  apply_surface( 383, 384, score_text, screen );
624 
625  //Background round se maggiore di 9

627  {

630  }
631 
632  //Testo round
633  round_text = TTF_RenderText_Solid( font, ( convert_round( round ) ).c_str(), green_font );
634  apply_surface( 80, 353, round_text, screen );
635 
636  //Msgbox round

650  //SDL_Flip( screen );
